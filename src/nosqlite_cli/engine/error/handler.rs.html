57    /// ðŸ¦€
58    /// Logs an error into the handler, optionally persisting it for later retrieval or storage.
59    ///
60    /// This method performs two actions:
61    /// 1. **Persists** the provided [`NosqliteError`] using the internal `persist_error` mechanism.
62    /// 2. **Stores** the error in the internal `errors` list for in-memory tracking.
63    ///
64    /// This allows errors to be:
65    /// - Programmatically inspected later (via `self.errors`)
66    /// - Persisted to disk or external logs (depending on `persist_error`'s implementation)
67    ///
68    /// # Parameters
69    ///
70    /// - `error`: The [`NosqliteError`] to log. This typically originates from collection operations
71    ///   such as schema validation failures, missing documents, or malformed input.
72    ///
73    /// # Example
74    ///
75    /// ```rust
76    /// use nosqlite_rust::engine::error::{NosqliteErrorHandler, NosqliteError};
77    /// let mut handler = NosqliteErrorHandler::new("temp/data13.nosqlite".to_string());
78    /// handler.log_error(NosqliteError::DocumentNotFound("abc123".to_string()));
79    /// assert_eq!(handler.all_errors().len(), 1);
80    /// ```
81    ///
82    /// # Side Effects
83    ///
84    /// - Calls `persist_error(&error)`, which may perform I/O or other logging operations.
85    ///
86    /// # See Also
87    ///
88    /// - [`NosqliteError`] â€” the error type being logged
89    pub fn log_error(&mut self, error: NosqliteError) {
90        let timestamp = chrono::Utc::now();
91        self.persist_error(&error);
92        self.errors.push(error);
93    }

95    /// ðŸ¦€
96    /// Persists a given [`NosqliteError`] to a log file associated with the database.
97    ///
98    /// This method appends the error, along with a UTC timestamp, to a `.log` file whose name
99    /// is derived from the handler's `db_path`. The log file is created if it does not already exist.
100    ///
101    /// # Log Format
102    ///
103    /// Each error is logged in the format:
104    /// ```text
105    /// [2025-04-10T18:03:12.456Z] DocumentNotFound("abc123")
106    /// ```
107    ///
108    /// # File Path
109    ///
110    /// The log file is derived from the `db_path` by replacing the `.nosqlite` extension with `.log`.
111    /// For example, `"data/users.nosqlite"` â†’ `"data/users.log"`
112    ///
113    /// # Parameters
114    ///
115    /// - `error`: A reference to the [`NosqliteError`] that will be written to the log.
116    ///
117    /// # Panics
118    ///
119    /// This method will **panic** if the log file cannot be opened or created, with the message:
120    /// `"Impossible d'ouvrir le fichier de log"`.
121    ///
122    /// # Side Effects
123    ///
124    /// - Appends a log entry to the target `.log` file
125    /// - Performs file I/O operations
126    ///
127    /// # Notes
128    ///
129    /// - This method is intended to be used internally by [`NosqliteErrorHandler::log_error`].
130    /// - Consider adding more robust error handling instead of a hard `expect()` in production environments.
131    ///
132    /// # See Also
133    ///
134    /// - [`NosqliteErrorHandler::log_error`] â€” logs and persists errors together
135    /// - [`NosqliteError`] â€” the structured error type
136    fn persist_error(&self, error: &NosqliteError) {
137        let log_path = self.db_path.replace(".nosqlite", ".log");
138
139        let mut file = std::fs::OpenOptions::new()
140            .create(true)
141            .append(true)
142            .open(log_path)
143            .expect("Impossible d'ouvrir le fichier de log");
144
145        let timestamp = chrono::Utc::now();
146        let entry = format!("[{}] {}\n", timestamp, error);
147
148        let _ = file.write_all(entry.as_bytes());
149    }

196    /// Wraps a fallible operation, logs any error using the handler, and returns a unified [`NosqliteError`].
197    ///
198    /// This utility method transforms a `Result<T, E>` into `Result<T, NosqliteError>`, using a provided closure
199    /// to convert the original error type `E` into a [`NosqliteError`]. If the result is an error, it will be:
200    /// - Transformed via the `wrap` closure
201    /// - Logged using [`NosqliteErrorHandler::log_error`]
202    /// - Returned to the caller as a `Result::Err`
203    ///
204    /// This method is especially useful for composing fallible operations in I/O, parsing, or collection logic,
205    /// without repeating error logging boilerplate.
206    ///
207    /// # Type Parameters
208    ///
209    /// - `T`: The success type of the incoming result.
210    /// - `E`: The error type to convert into a [`NosqliteError`].
211    ///
212    /// # Parameters
213    ///
214    /// - `result`: The `Result<T, E>` to evaluate.
215    /// - `wrap`: A closure used to convert the error `E` into a [`NosqliteError`].
216    ///
217    /// # Returns
218    ///
219    /// - `Ok(T)` if the original result is successful.
220    /// - `Err(NosqliteError)` if the original result is an error, with logging side-effects.
221    ///
222    /// # Example
223    ///
224    /// ```rust
225    /// use nosqlite_rust::engine::error::{NosqliteErrorHandler, NosqliteError};
226    ///
227    /// let result: Result<i32, std::io::Error> = Err(std::io::Error::new(std::io::ErrorKind::Other, "fail"));
228    ///
229    /// let mut handler = NosqliteErrorHandler::new("temp/data15.nosqlite".to_string());
230    /// let mapped = handler.try_or_log(result, |e| NosqliteError::IoError(format!("IO failed: {}", e)));
231    ///
232    /// assert!(mapped.is_err());
233    /// assert_eq!(handler.all_errors().len(), 1);
234    /// ```
235    ///
236    /// # See Also
237    ///
238    /// - [`NosqliteErrorHandler::log_error`] â€” used internally to record the error
239    /// - [`NosqliteError`] â€” the target error type
240    pub fn try_or_log<T, E>(
241        &mut self,
242        result: Result<T, E>,
243        wrap: impl FnOnce(E) -> NosqliteError,
244    ) -> Result<T, NosqliteError> {
245        result.map_err(|e| {
246            let err = wrap(e);
247            self.log_error(err.clone());
248            err
249        })
250    }

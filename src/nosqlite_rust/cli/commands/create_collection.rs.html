5/// Parses and executes the `db.createCollection()` command from the CLI,
6/// extracting the collection name and optional schema from the input.
7///
8/// This function does the following:
9/// - Parses the input to extract the collection name and an optional JSON schema
10/// - Falls back to an empty schema (`{}`) if no schema is provided
11/// - Calls [`Nosqlite::create_collection`] to create and register the collection in the database
12///
13/// # Parameters
14///
15/// - `input`: A string containing the CLI command to create the collection. Must follow the format:
16///   - `db.createCollection("name")`
17///   - `db.createCollection("name", {schema})`
18/// - `db`: A mutable reference to the [`Nosqlite`] instance where the collection should be added.
19///
20/// # Returns
21///
22/// - `Ok(String)` containing a success message if the collection is created successfully.
23/// - `Err(String)` describing the error if parsing fails or collection creation fails.
24///
25/// # Errors
26///
27/// - Returns a string describing:
28///   - Syntax errors (e.g. missing closing parenthesis or name)
29///   - Invalid JSON in the schema
30///   - Errors returned by the internal `create_collection` call
31///
32/// # See Also
33///
34/// - [`todo!`]
35pub fn handle_create_collection(input: &str, db: &mut Nosqlite) -> Result<String, String> {
36    let (name, schema) = parse_command_args(input)?;
37    db.create_collection(name, schema)
38        .map(|_| format!("Collection '{}' created successfully", name))
39        .map_err(|e| format!("Failed to create collection: {e}"))
40}

42fn parse_command_args(input: &str) -> Result<(&str, Value), String> {
43    let args = input
44        .strip_prefix("db.createCollection(")
45        .and_then(|s| s.strip_suffix(')'))
46        .ok_or_else(|| "Syntax error: missing closing ')'".to_string())?;
47
48    let mut parts = args.splitn(2, ',').map(str::trim);
49
50    let name_raw = parts
51        .next()
52        .ok_or_else(|| "Syntax error: collection name missing".to_string())?
53        .trim_matches(|c| c == '"' || c == '\'');
54
55    let schema = match parts.next() {
56        Some(json_str) => {
57            let json_str = json_str.replace('\'', "\"");
58            serde_json::from_str(&json_str).map_err(|_| "Invalid JSON schema".to_string())?
59        }
60        None => Value::Object(serde_json::Map::new()),
61    };
62    Ok((name_raw, schema))
63}

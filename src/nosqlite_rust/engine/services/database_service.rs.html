8/// Loads a [`Database`] instance from disk, or creates a new one if the file doesn't exist.
9///
10/// This high-level helper wraps the internal [`File::load_or_create`] function, providing a convenient
11/// way to bootstrap your NoSQL engine from persistent storage. If the target file:
12///
13/// - **Exists**:
14///   - Attempts to read and decrypt it
15///   - Deserializes the contents into a [`Database`] instance
16///
17/// - **Does not exist**:
18///   - Creates a new empty [`Database`] with the specified path
19///
20/// Any encountered errors are logged using the provided [`NosqliteErrorHandler`] and returned.
21///
22/// # Parameters
23///
24/// - `path`: The file path to the encrypted `.nosqlite` database file.
25/// - `error_handler`: A mutable reference to a [`NosqliteErrorHandler`] for structured logging.
26///
27/// # Returns
28///
29/// - `Ok(Database)` if the file is successfully read and parsed, or a new instance is created
30/// - `Err(NosqliteError)` if the file cannot be read, decrypted, or deserialized
31///
32/// # Example
33///
34/// ```rust
35/// use nosqlite_rust::engine::models::database::model::Database;
36/// use nosqlite_rust::engine::error::{NosqliteErrorHandler, NosqliteError};
37/// use nosqlite_rust::engine::services::database_service::load_or_create_database;
38///
39/// let mut handler = NosqliteErrorHandler::new("temp/data20.nosqlite".to_string());
40/// let db = load_or_create_database("temp/data20.nosqlite", &mut handler)?;
41/// println!("{}", db);
42/// Ok::<(), NosqliteError>(())
43/// ```
44///
45/// # Notes
46///
47/// - This function handles decryption, format validation, and automatic creation in one call.
48/// - Uses internal AES-256-GCM encryption via the [`File`] module.
49///
50/// # See Also
51///
52/// - [`save_database`] â€” for persisting the database back to disk
53/// - [`File::load_or_create`] â€” internal loader implementation
54/// - [`NosqliteError`] â€” structured error type
55pub fn load_or_create_database(
56    path: &str,
57    error_handler: &mut NosqliteErrorHandler,
58) -> Result<Database, NosqliteError> {
59    match File::load_or_create(path, error_handler) {
60        Ok(db) => Ok(db),
61        Err(e) => {
62            error_handler.log_error(e.clone());
63            Err(e)
64        }
65    }
66}

69/// Saves the current state of the database to disk at the specified path.
70///
71/// This function wraps [`File::save`] to serialize and securely encrypt the [`Database`] contents,
72/// then write them to the filesystem. Any errors during serialization, encryption, or I/O
73/// are logged using the provided error handler.
74///
75/// # Parameters
76///
77/// - `path`: The file path to write the encrypted database to.
78/// - `db`: A reference to the [`Database`] instance to be saved.
79/// - `error_handler`: A mutable reference to a [`NosqliteErrorHandler`] for logging any I/O or serialization errors.
80///
81/// # Returns
82///
83/// - `Ok(())` if the database is successfully saved
84/// - `Err(NosqliteError)` if serialization, encryption, or file write fails
85///
86/// # Example
87///
88/// ```rust
89/// use nosqlite_rust::engine::models::database::model::Database;
90/// use nosqlite_rust::engine::error::{NosqliteErrorHandler, NosqliteError};
91/// use nosqlite_rust::engine::services::database_service::save_database;
92///
93/// let db = Database::new("temp/data21.nosqlite");
94/// let mut handler = NosqliteErrorHandler::new("temp/data21.nosqlite".to_string());
95/// save_database("temp/data21.nosqlite", &db, &mut handler)?;
96/// println!("Database saved successfully.");
97/// Ok::<(), NosqliteError>(())
98/// ```
99///
100/// # Notes
101///
102/// - Uses AES-256-GCM encryption via the [`File`] module.
103/// - Existing files will be **overwritten**. Make sure to backup if needed.
104///
105/// # See Also
106///
107/// - [`load_or_create_database`] â€” for the reverse operation
108/// - [`File::save`] â€” internal save logic
109/// - [`NosqliteError`] â€” error enum
110pub fn save_database(
111    path: &str,
112    db: &Database,
113    error_handler: &mut NosqliteErrorHandler,
114) -> Result<(), NosqliteError> {
115    match File::save(path, db, error_handler) {
116        Ok(_) => Ok(()),
117        Err(e) => {
118            error_handler.log_error(e.clone());
119            Err(e)
120        }
121    }

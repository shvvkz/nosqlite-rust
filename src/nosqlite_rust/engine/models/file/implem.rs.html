15    /// Loads the database from disk, or creates a new one if the file does not exist.
16    ///
17    /// This function attempts to read, decrypt, and deserialize a database file into a [`Database`] instance.
18    /// If the file does not exist, a new in-memory database is created using the specified path.
19    ///
20    /// It also ensures a valid encryption key is available by attempting to load or generate one
21    /// from the default key path (`DEFAULT_KEY_PATH`).
22    ///
23    /// # Parameters
24    ///
25    /// - `db_path`: The filesystem path to the encrypted `.nosqlite` database file.
26    /// - `handler`: A mutable reference to a [`NosqliteErrorHandler`] for structured logging and error tracking.
27    ///
28    /// # Returns
29    ///
30    /// - `Ok(Database)` if the database was successfully loaded or created.
31    /// - `Err(NosqliteError)` if any stage of key loading, file reading, decryption, or deserialization fails.
32    ///
33    /// # Behavior
34    ///
35    /// - If the file **exists**:
36    ///   1. Reads its contents as an encrypted string
37    ///   2. Attempts decryption using the AES key from `DEFAULT_KEY_PATH`
38    ///   3. Attempts deserialization of the decrypted JSON into a [`Database`]
39    ///
40    /// - If the file **does not exist**:
41    ///   - Returns a new `Database` instance with no collections
42    ///
43    /// # Panics
44    ///
45    /// This method **may panic** only in unrecoverable conditions such as:
46    /// - Inability to open the database file (e.g. permissions issues not caught earlier)
47    /// - Internal `expect()`/`unwrap()` usage in key/decryption layers
48    ///
49    /// # Example
50    ///
51    /// ```rust
52    /// use nosqlite_rust::engine::error::{NosqliteErrorHandler, NosqliteError};
53    /// use nosqlite_rust::engine::models::File;
54    ///
55    /// let mut handler = NosqliteErrorHandler::new("temp/data42.nosqlite".to_string());
56    /// let db = File::load_or_create("temp/data42.nosqlite", &mut handler)?;
57    /// println!("{}", db);
58    /// Ok::<(), NosqliteError>(())
59    /// ```
60    ///
61    /// # Security Notes
62    ///
63    /// - Encryption is handled via `File::decrypt`
64    /// - Keys are managed via `File::load_or_generate_key` and stored at `DEFAULT_KEY_PATH`
65    /// - Ensure secure handling of key and file paths in production deployments
66    ///
67    /// # See Also
68    ///
69    /// - [`Database::new`] â€” creates a new empty database
70    /// - [`NosqliteErrorHandler`] â€” handles structured error logging
71    /// - [`NosqliteError`] â€” error type used throughout the system
72    pub fn load_or_create(
73        db_path: &str,
74        handler: &mut NosqliteErrorHandler,
75    ) -> Result<Database, NosqliteError> {
76        let key = Self::load_or_generate_key(DEFAULT_KEY_PATH, handler)?;
77
78        if Path::new(db_path).exists() {
79            let encrypted = handler.try_or_log(fs::read_to_string(db_path), |e| {
80                NosqliteError::IoError(e.to_string())
81            })?;
82            let decrypted_result = Self::decrypt(&encrypted, &key, handler);
83            let decrypted = handler.try_or_log(decrypted_result, |e| {
84                NosqliteError::EncryptionError(e.to_string())
85            })?;
86
87            let db: Database = handler
88                .try_or_log(serde_json::from_str(&decrypted), |e| {
89                    NosqliteError::DeserializationError(e.to_string())
90                })
91                .map_err(|_| {
92                    let err = NosqliteError::InvalidDatabaseFormat(
93                        "Failed to deserialize database".to_string(),
94                    );
95                    handler.log_error(err.clone());
96                    err
97                })?;
98            Ok(db)
99        } else {
100            Ok(Database::new(db_path))
101        }
102    }

106    ///
107    /// This method serializes the entire database to pretty-printed JSON, encrypts it
108    /// using AES-256-GCM, and writes the result to the provided path. It ensures that all
109    /// saved data remains secure and tamper-resistant.
110    ///
111    /// # Parameters
112    ///
113    /// - `db_path`: The file path where the encrypted database should be written.
114    /// - `db`: The [`Database`] instance to be serialized and persisted.
115    /// - `handler`: A mutable reference to a [`NosqliteErrorHandler`] for structured logging of any I/O,
116    ///   serialization, or encryption errors.
117    ///
118    /// # Returns
119    ///
120    /// - `Ok(())` if the database was successfully saved
121    /// - `Err(NosqliteError)` if any step in the pipeline fails
122    ///
123    /// # Pipeline Steps
124    ///
125    /// 1. **Key management**: Loads or generates an AES encryption key via `load_or_generate_key`
126    /// 2. **Serialization**: Converts the `Database` instance to pretty-printed JSON
127    /// 3. **Encryption**: Encrypts the JSON payload with AES-256-GCM
128    /// 4. **File write**: Saves the encrypted data to `db_path`
129    ///
130    /// # Panics
131    ///
132    /// While this method uses structured error logging (`try_or_log`) throughout, it may still
133    /// panic in rare cases (e.g. internal `expect()` calls in utility methods).
134    ///
135    /// # Example
136    ///
137    /// ```rust
138    /// use nosqlite_rust::engine::models::{Database, File};
139    /// use nosqlite_rust::engine::error::NosqliteErrorHandler;
140    ///
141    /// let db = Database::default();
142    /// let mut handler = NosqliteErrorHandler::new("temp/data43.nosqlite".to_string());
143    ///
144    /// File::save("temp/data43.nosqlite", &db, &mut handler).expect("Failed to save database");
145    /// ```
146    ///
147    /// # Security Notes
148    ///
149    /// - Encryption is performed using AES-256-GCM with keys stored at `DEFAULT_KEY_PATH`
150    /// - The resulting file at `db_path` is **not human-readable**
151    /// - For plaintext backups or inspection, implement a `save_as_json()` variant
152    ///
153    /// # See Also
154    ///
155    /// - [`File::load_or_create`] â€” for loading the encrypted database back
156    /// - [`NosqliteError`] â€” unified error type
157    /// - [`NosqliteErrorHandler`] â€” used for logging any failures in the pipeline
158    pub fn save(
159        db_path: &str,
160        db: &Database,
161        handler: &mut NosqliteErrorHandler,
162    ) -> Result<(), NosqliteError> {
163        let key = Self::load_or_generate_key(DEFAULT_KEY_PATH, handler)?;
164        let json = handler.try_or_log(serde_json::to_string_pretty(db), |e| {
165            NosqliteError::SerializationError(e.to_string())
166        })?;
167        let encrypted_result = Self::encrypt(&json, &key, handler);
168        let encrypted = handler.try_or_log(encrypted_result, |e| {
169            NosqliteError::EncryptionError(e.to_string())
170        })?;
171        handler.try_or_log(fs::write(db_path, &encrypted), |e| {
172            NosqliteError::IoError(e.to_string())
173        })?;
174        Ok(())
175    }

178    /// Encrypts a plaintext string using AES-256-GCM and returns a base64-encoded result.
179    ///
180    /// This method performs authenticated encryption using the AES-256-GCM algorithm. The resulting
181    /// ciphertext is prepended with the generated 12-byte nonce and encoded using base64. This format
182    /// is compact and easy to store or transmit.
183    ///
184    /// # Parameters
185    ///
186    /// - `data`: The plaintext string to encrypt.
187    /// - `key`: A 256-bit AES key (`[u8; 32]`) used for encryption. This must match the key
188    ///   used during decryption.
189    /// - `handler`: A mutable reference to a [`NosqliteErrorHandler`] used to log any encryption failure.
190    ///
191    /// # Returns
192    ///
193    /// - `Ok(String)`: A base64-encoded string containing `[nonce || ciphertext]`
194    /// - `Err(NosqliteError::EncryptionError)`: If encryption fails internally
195    ///
196    /// # Format
197    ///
198    /// The returned string encodes:
199    /// ```text
200    /// [12-byte nonce][ciphertext] -> base64
201    /// ```
202    /// The nonce is randomly generated per encryption and included in the output for
203    /// later decryption.
204    ///
205    /// # Security Notes
206    ///
207    /// - A unique random nonce is generated for every call to `encrypt`
208    /// - Do **not reuse** the same nonce with the same key across different messages
209    /// - This method uses AES-256-GCM from the [`aes-gcm`] crate (authenticated encryption)
210    ///
211    /// # See Also
212    ///
213    /// - [`Database::decrypt`] â€” decrypts output from this method
214    /// - [`NosqliteError`] â€” error enum for encryption/decryption failures
215    /// - [`Aes256Gcm`] â€” the underlying cipher
216    fn encrypt(
217        data: &str,
218        key: &[u8; 32],
219        handler: &mut NosqliteErrorHandler,
220    ) -> Result<String, NosqliteError> {
221        let cipher = Aes256Gcm::new(Key::<Aes256Gcm>::from_slice(key));
222        let nonce = Aes256Gcm::generate_nonce(&mut OsRng); // 12 bytes
223        let ciphertext = handler.try_or_log(cipher.encrypt(&nonce, data.as_bytes()), |e| {
224            NosqliteError::EncryptionError(e.to_string())
225        })?;
226
227        let mut result = nonce.to_vec();
228        result.extend(ciphertext);
229        Ok(general_purpose::STANDARD.encode(&result))
230    }

232    /// ðŸ¦€
233    /// Decrypts a base64-encoded encrypted string using AES-256-GCM.
234    ///
235    /// This method reverses the process performed by [`encrypt`]. It:
236    /// 1. Base64-decodes the input string
237    /// 2. Extracts the 12-byte nonce from the beginning of the byte stream
238    /// 3. Decrypts the remaining ciphertext using AES-256-GCM
239    /// 4. Converts the plaintext bytes back into a UTF-8 string
240    ///
241    /// # Parameters
242    ///
243    /// - `data`: A base64-encoded string produced by the [`encrypt`] method. Must contain
244    ///   both the nonce and ciphertext in the correct order.
245    /// - `key`: A 256-bit AES decryption key (`[u8; 32]`). This must match the key used during encryption.
246    /// - `handler`: A mutable reference to a [`NosqliteErrorHandler`] for logging any decode/decrypt errors.
247    ///
248    /// # Returns
249    ///
250    /// - `Ok(String)`: The successfully decrypted plaintext string
251    /// - `Err(NosqliteError)`: If decoding, decryption, or UTF-8 conversion fails
252    ///
253    /// # Format Assumptions
254    ///
255    /// The input string must have been generated by [`Database::encrypt`] and should follow:
256    /// ```text
257    /// base64([12-byte nonce][ciphertext])
258    /// ```
259    ///
260    /// # Security Notes
261    ///
262    /// - Decryption is **authenticated** â€” tampering with the ciphertext or nonce will cause it to fail
263    /// - The nonce is extracted from the first 12 bytes of the decoded input
264    /// - AES-256-GCM provides both confidentiality and integrity
265    ///
266    /// # See Also
267    ///
268    /// - [`Database::encrypt`] â€” encrypts plaintext using AES-256-GCM
269    /// - [`NosqliteError`] â€” contains possible decryption and decoding error variants
270    /// - [`general_purpose::STANDARD`] â€” the base64 variant used
271    fn decrypt(
272        data: &str,
273        key: &[u8; 32],
274        handler: &mut NosqliteErrorHandler,
275    ) -> Result<String, NosqliteError> {
276        let decoded = handler.try_or_log(general_purpose::STANDARD.decode(data), |e| {
277            NosqliteError::Base64DecodeError(e.to_string())
278        })?;
279        let (nonce_bytes, ciphertext) = decoded.split_at(12);
280        let cipher = Aes256Gcm::new(Key::<Aes256Gcm>::from_slice(key));
281        let nonce = Nonce::from_slice(nonce_bytes);
282
283        let plaintext = handler.try_or_log(cipher.decrypt(nonce, ciphertext), |e| {
284            NosqliteError::EncryptionError(e.to_string())
285        })?;
286        let decrypt = handler.try_or_log(String::from_utf8(plaintext.clone()), |e| {
287            NosqliteError::DeserializationError(e.to_string())
288        })?;
289        Ok(decrypt)
290    }

292    /// ðŸ¦€
293    /// Loads a 256-bit AES encryption key from a file, or generates and stores a new one if it doesnâ€™t exist.
294    ///
295    /// This method ensures that a valid 32-byte key is always available for encryption and decryption.
296    /// It checks for the existence of the key file:
297    ///
298    /// - If the file **exists**:
299    ///   - Reads it as a hex-encoded string
300    ///   - Decodes it into a 256-bit key (`[u8; 32]`)
301    ///
302    /// - If the file **does not exist**:
303    ///   - Generates a new secure random 256-bit key
304    ///   - Encodes it as a hex string
305    ///   - Writes it to disk at the specified `path`
306    ///
307    /// # Parameters
308    ///
309    /// - `path`: The path to the key file to load or create.
310    /// - `handler`: A mutable reference to a [`NosqliteErrorHandler`] used to log any I/O or decoding errors.
311    ///
312    /// # Returns
313    ///
314    /// - `Ok([u8; 32])` â€” a 256-bit AES key loaded or generated successfully
315    /// - `Err(NosqliteError)` â€” if any step (file I/O, decoding, or writing) fails
316    ///
317    /// # Behavior
318    ///
319    /// - The generated key is encoded in lowercase hexadecimal (64 characters) before being written to disk.
320    /// - The same key must be used for both encryption and decryption.
321    /// - Ensures idempotent key access: same file, same key across reboots.
322    ///
323    /// # Panics
324    ///
325    /// This method panics **only if** the random number generator or file system encounters
326    /// unrecoverable issues not caught by `try_or_log` (e.g. path permission failure + missing error log).
327    ///
328    /// # Security Notes
329    ///
330    /// - The generated key should be kept secret and backed up securely.
331    /// - Avoid sharing the same key across different databases unless intentional.
332    /// - Do not edit the key file manually unless you're restoring from a trusted backup.
333    ///
334    /// # See Also
335    ///
336    /// - [`Database::encrypt`] / [`Database::decrypt`] â€” uses this key
337    /// - [`NosqliteErrorHandler`] â€” for persistent error logging
338    /// - [`rand::rng`] â€” used to generate secure random bytes
339    fn load_or_generate_key(
340        path: &str,
341        handler: &mut NosqliteErrorHandler,
342    ) -> Result<[u8; 32], NosqliteError> {
343        if Path::new(path).exists() {
344            let content = handler.try_or_log(fs::read_to_string(path), |e| {
345                NosqliteError::IoError(e.to_string())
346            })?;
347            let bytes = handler.try_or_log(hex::decode(content.trim()), |e| {
348                NosqliteError::HexDecodeError(e.to_string())
349            })?;
350            let mut key = [0u8; 32];
351            key.copy_from_slice(&bytes[..32]);
352            Ok(key)
353        } else {
354            use rand::RngCore;
355            let mut raw = [0u8; 32];
356            rand::rng().fill_bytes(&mut raw);
357            let hex = hex::encode(raw);
358            handler.try_or_log(fs::write(path, hex), |e| {
359                NosqliteError::IoError(e.to_string())
360            })?;
361            Ok(raw)
362        }
363    }

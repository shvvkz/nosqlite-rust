6/// Returns the current Unix timestamp in seconds.
7///
8/// This utility function is used throughout the system to timestamp entities such as:
9/// - Document creation (`created_at`)
10/// - Document updates (`updated_at`)
11/// - Collection initialization
12///
13/// It calculates the number of seconds elapsed since the Unix epoch (`1970-01-01T00:00:00Z`)
14/// using the system clock.
15///
16/// # Returns
17///
18/// - A `u64` representing the number of seconds since the Unix epoch.
19///
20/// # Example
21///
22/// ```rust
23/// use nosqlite_rust::engine::models::utils::now;
24///
25/// let timestamp = now();
26/// println!("Current UNIX timestamp: {}", timestamp);
27/// ```
28///
29/// # Panics
30///
31/// This function will **panic** if the system clock is set before the Unix epoch,
32/// which is highly unlikely on modern systems.
33///
34/// # Notes
35///
36/// - This function is timezone-independent (always UTC)
37/// - Consider formatting timestamps for human output using a library like `chrono`
38///
39/// # See Also
40///
41/// - [`SystemTime::now`] â€” standard library source of time
42/// - [`UNIX_EPOCH`] â€” constant used for offset
43/// - [`Document::new`] â€” sets both `created_at` and `updated_at` using this
44pub fn now() -> u64 {
45    SystemTime::now()
46        .duration_since(UNIX_EPOCH)
47        .unwrap()
48        .as_secs()
49}

51/// ðŸ¦€
52/// Validates a JSON document against a given schema definition.
53///
54/// This function performs recursive structural validation of a document based on a JSON schema.
55/// The schema is defined as a map (`serde_json::Map`) where:
56/// - Keys represent field names
57/// - Values define the expected type as a string (e.g. `"string"`, `"number"`, `"boolean"`)
58///   or a nested JSON object (representing a sub-schema for nested documents)
59///
60/// All fields defined in the schema must be present in the document, and their types must match.
61/// Extra fields in the document are allowed (non-strict mode).
62///
63/// # Parameters
64///
65/// - `doc`: A reference to the document (as a JSON object) to validate.
66/// - `schema`: A reference to the expected schema structure (also a JSON object).
67///
68/// # Returns
69///
70/// - `true` if the document fully conforms to the schema.
71/// - `false` if any required field is missing or if any type mismatches occur.
72///
73/// # Supported Types
74///
75/// - `"string"`
76/// - `"number"`
77/// - `"boolean"`
78/// - `"array"`
79/// - `"object"`
80///
81/// # Example
82///
83/// ```rust
84/// use serde_json::json;
85/// use std::collections::BTreeMap;
86/// use nosqlite_rust::engine::models::utils::validate_against_structure;
87///
88/// let schema = json!({
89///     "title": "string",
90///     "views": "number",
91///     "meta": {
92///         "tags": "array",
93///         "published": "boolean"
94///     }
95/// }).as_object().unwrap().clone();
96///
97/// let doc = json!({
98///     "title": "My Post",
99///     "views": 120,
100///     "meta": {
101///         "tags": ["rust", "nosql"],
102///         "published": true
103///     }
104/// }).as_object().unwrap().clone();
105///
106/// assert!(validate_against_structure(&doc, &schema));
107/// ```
108///
109/// # Notes
110///
111/// - The function performs **recursive** validation for nested objects.
112/// - Only fields present in the schema are checked; extra fields in the document are ignored.
113/// - Type matching is handled internally by `type_matches` helper function.
114///
115/// # See Also
116///
117/// - [`serde_json::Value`] â€” JSON representation used in both document and schema
118pub fn validate_against_structure(
119    doc: &serde_json::Map<String, Value>,
120    schema: &serde_json::Map<String, Value>,
121) -> bool {
122    for (key, expected_type) in schema {
123        match doc.get(key) {
124            Some(actual_value) => match (expected_type, actual_value) {
125                (Value::String(s_type), val) => {
126                    if !type_matches(s_type, val) {
127                        return false;
128                    }
129                }
130                (Value::Object(sub_schema), Value::Object(sub_doc)) => {
131                    if !validate_against_structure(sub_doc, sub_schema) {
132                        return false;
133                    }
134                }
135                _ => return false,
136            },
137            None => return false,
138        }
139    }
140    true
141}

143/// ðŸ¦€
144/// Checks if a JSON value matches the expected type (case-insensitive).
145///
146/// This helper function performs a basic type check between a user-defined type string
147/// and a [`serde_json::Value`]. It supports simple, high-level JSON data types and is
148/// intended for use in schema validation (e.g., [`validate_against_structure`]).
149///
150/// # Parameters
151///
152/// - `expected`: A case-insensitive string representing the expected type.
153/// - `val`: The actual [`Value`] to be checked.
154///
155/// # Supported Types
156///
157/// - `"string"` â†’ [`Value::is_string()`]
158/// - `"number"` â†’ [`Value::is_number()`]
159/// - `"boolean"` â†’ [`Value::is_boolean()`]
160/// - `"array"` â†’ [`Value::is_array()`]
161/// - `"object"` â†’ [`Value::is_object()`]
162///
163/// # Returns
164///
165/// - `true` if the value matches the expected type
166/// - `false` if the types are incompatible or the type string is unrecognized
167///
168/// # Notes
169///
170/// - Type names are matched case-insensitively (`"String"` and `"string"` are equivalent)
171/// - Unrecognized type strings return `false`
172///
173/// # See Also
174///
175/// - [`validate_against_structure`] â€” uses this function for document schema enforcement
176/// - [`serde_json::Value`] â€” the dynamic JSON type used for validation
177fn type_matches(expected: &str, val: &Value) -> bool {
178    match expected.to_lowercase().as_str() {
179        "string" => val.is_string(),
180        "number" => val.is_number(),
181        "boolean" => val.is_boolean(),
182        "array" => val.is_array(),
183        "object" => val.is_object(),
184        _ => false,
185    }
186}

188/// ðŸ¦€
189/// Retrieves a nested value from a JSON object using dot-separated path syntax.
190///
191/// This utility function traverses a [`serde_json::Value`] using a field path such as `"user.profile.name"`
192/// and returns the corresponding nested value, if it exists.
193///
194/// # Parameters
195///
196/// - `value`: A reference to a [`serde_json::Value`], typically a JSON object or sub-object.
197/// - `path`: A string slice representing the dot-separated path to the desired field (e.g., `"address.city"`).
198///
199/// # Returns
200///
201/// - `Some(&Value)` if the full path exists and points to a valid value.
202/// - `None` if any part of the path is missing or not an object.
203///
204/// # Example
205///
206/// ```rust
207/// use serde_json::json;
208/// use nosqlite_rust::engine::models::utils::get_nested_value;
209///
210/// let data = json!({
211///     "user": {
212///         "profile": {
213///             "name": "Alice"
214///         }
215///     }
216/// });
217///
218/// assert_eq!(
219///     get_nested_value(&data, "user.profile.name"),
220///     Some(&json!("Alice"))
221/// );
222///
223/// assert_eq!(get_nested_value(&data, "user.profile.age"), None);
224/// ```
225///
226/// # Notes
227///
228/// - This function does not support array indexing (e.g., `"items[0].name"` is not valid).
229/// - Useful for filtering or querying deeply nested document fields.
230pub fn get_nested_value<'a>(value: &'a Value, path: &str) -> Option<&'a Value> {
231    path.split('.').try_fold(value, |val, key| val.get(key))
232}
